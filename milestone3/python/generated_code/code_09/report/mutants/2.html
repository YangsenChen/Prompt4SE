class Organization:

        self.name = name
        self.short_name = short_name
        self.description = description
        self.logo = logo
    
    def __str__(self):
        return f'Organization(id={self.id}, name={self.name}, short_name={self.short_name}, description={self.description}, logo={self.logo})'
